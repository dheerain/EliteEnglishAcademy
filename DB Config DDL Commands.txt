CREATE TABLE STUDENT(
    STUDENT_ID varchar(255) NOT NULL UNIQUE, 
    NUMBER INT NOT NULL  PRIMARY KEY AUTO_INCREMENT,
    NAME varchar(255),
    FATHERS_NAME varchar(255),
    MOTHERS_NAME varchar(255),
    ADDRESS varchar(255),
    MOBILE_NO VARCHAR(25),
    ALT_MOBILE_NO VARCHAR(25),
    GENDER VARCHAR(10),
    DATE_OF_BIRTH DATE,
    EMAIL VARCHAR(255),
    NAME_OF_LAST_SCHOOL VARCHAR(255),
    ENROLLED_IN VARCHAR(10),
    CONTRACT_STATUS VARCHAR(10),
    CONTRACT_START_DATE DATE,
    CONTRACT_TERMINATION_DATE DATE,
    BANDS_RECEIVED FLOAT
);
